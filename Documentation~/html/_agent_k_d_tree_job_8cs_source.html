54 private void BuildAgentTreeRecursive(int begin, int end, int node)
55 {
56
57 AgentTreeNode treeNode = m_outputTree[node];
58 AgentData agent = m_inputAgents[begin];
59 float2 pos;
60
61 treeNode.begin = begin;
62 treeNode.end = end;
63 treeNode.minX = treeNode.maxX = agent.position.x;
64 treeNode.minY = treeNode.maxY = agent.position.y;
65
66 for (int i = begin + 1; i < end; ++i)
67 {
68 pos = m_inputAgents[i].position;
69 treeNode.maxX = max(treeNode.maxX, pos.x);
70 treeNode.minX = min(treeNode.minX, pos.x);
71 treeNode.maxY = max(treeNode.maxY, pos.y);
72 treeNode.minY = min(treeNode.minY, pos.y);
73 }
74
75 m_outputTree[node] = treeNode;
76
77 if (end - begin > AgentTreeNode.MAX_LEAF_SIZE)
78 {
79 // No leaf node.
80 bool isVertical = treeNode.maxX - treeNode.minX > treeNode.maxY - treeNode.minY;
81 float splitValue = 0.5f * (isVertical ? treeNode.maxX + treeNode.minX : treeNode.maxY + treeNode.minY);
82
83 int left = begin;
84 int right = end;
85
86 while (left < right)
87 {
88 while (left < right && (isVertical ? m_inputAgents[left].position.x : m_inputAgents[left].position.y) < splitValue)
89 {
90 ++left;
91 }
92
93 while (right > left && (isVertical ? m_inputAgents[right - 1].position.x : m_inputAgents[right - 1].position.y) >= splitValue)
94 {
95 --right;
96 }
97
98 if (left < right)
99 {
100 AgentData tempAgent = m_inputAgents[left], rep = m_inputAgents[right - 1];
101 tempAgent.kdIndex = right - 1;
102 rep.kdIndex = left;
103 m_inputAgents[left] = rep;
104 m_inputAgents[right - 1] = tempAgent;
105 ++left;
106 --right;
107 }
108 }
109
110 int leftSize = left - begin;
111
112 if (leftSize == 0)
113 {
114 ++leftSize;
115 ++left;
116 ++right;
117 }
118
119 treeNode.left = node + 1;
120 treeNode.right = node + 2 * leftSize;
121 m_outputTree[node] = treeNode;
122
123 BuildAgentTreeRecursive(begin, left, treeNode.left);
124 BuildAgentTreeRecursive(left, end, treeNode.right);
125 }
126 }
