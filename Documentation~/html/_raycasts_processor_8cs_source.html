121 protected override void Apply(ref RaycastsJob job)
122 {
123 //Update raycast data object.
124 List<Raycast> lockedRaycasts = m_raycastProvider.lockedRaycasts;
125 Raycast ray;
126
127 RaycastData rayData;
128 RaycastResult rayResult;
129
130 Obstacle o;
131
132 int index = -1;
133
134 for (int i = 0, rayCount = lockedRaycasts.Count; i < rayCount; i++)
135 {
136 ray = lockedRaycasts[i];
137 ray.m_anyHit = false;
138
139 rayData = job.m_inputRaycasts[i];
140 rayResult = job.m_results[i];
141
142 //Agent hit
143 index = rayResult.hitAgent;
144 if (index != -1)
145 {
146
147 ray.agentHit = m_agentProvider.lockedAgents[index];
148 ray.agentHitLocation = rayResult.hitAgentLocation;
149 ray.m_anyHit = true;
150
151 }
152 else{ ray.agentHit = null; }
153
154 //Obstacle hit
155 index = rayResult.hitObstacle;
156 if(index != -1)
157 {
158
159 if (!rayResult.dynamicObstacle)
160 o = m_staticObstaclesProvider.obstacles[job.m_staticObstacleInfos[index].index];
161 else
162 o = m_dynObstaclesProvider.obstacles[job.m_dynObstacleInfos[index].index];
163
164 ray.obstacleHit = o;
165 ray.obstacleHitLocation = rayResult.hitObstacleLocation;
166 ray.m_anyHit = true;
167
168 }
169 else{ ray.obstacleHit = null; }
170
171 }
