60 int count = m_raycasts == null ? 0 : m_raycasts.Count;

69 protected override void Prepare(ref Unemployed job, float delta)
70 {
71
72 int raycastCount = m_lockedRaycasts.Count;
73
74 if (m_outputRaycast.Length != raycastCount)
75 {
76 m_outputRaycast.Dispose();
77 m_outputRaycast = new NativeArray<RaycastData>(raycastCount, Allocator.Persistent);
78 }
79
80 Raycast r;
81 float3 pos, dir;
82
83 if (plane == AxisPair.XY)
84 {
85 for (int i = 0; i < raycastCount; i++)
86 {
87 r = m_lockedRaycasts[i];
88 pos = r.pos;
89 dir = r.m_dir;
90 m_outputRaycast[i] = new RaycastData()
91 {
92 position = float2(pos.x, pos.y), //
93 distance = r.m_distance,
94 worldPosition = pos,
95 baseline = pos.z,
96 direction = float2(dir.x, dir.y),
97 worldDir = dir,
98 layerIgnore = r.m_layerIgnore,
99 filter = r.m_filter,
100 twoSided = r.twoSided
101 };
102 }
103 }
104 else
105 {
106 for (int i = 0; i < raycastCount; i++)
107 {
