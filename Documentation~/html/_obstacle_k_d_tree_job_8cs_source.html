68 private void BuildAgentTreeRecursive(int begin, int end, int node)
69 {
70
71 ObstacleTreeNode treeNode = m_outputTree[node];
72 ObstacleVertexData obstacle = m_inputObstacles[begin];
73 float2 pos;
74 float minX, minY, maxX, maxY;
75
76 treeNode.begin = begin;
77 treeNode.end = end;
78 minX = maxX = obstacle.pos.x;
79 minY = maxY = obstacle.pos.y;
80
81 for (int i = begin + 1; i < end; ++i)
82 {
83 pos = m_inputObstacles[i].pos;
84 maxX = max(maxX, pos.x);
85 minX = min(minX, pos.x);
86 maxY = max(maxY, pos.y);
87 minY = min(minY, pos.y);
88 }
89
90 treeNode.minX = minX;
91 treeNode.maxX = maxX;
92 treeNode.minY = minY;
93 treeNode.maxY = maxY;
94
95 m_outputTree[node] = treeNode;
96
97 if (end - begin > ObstacleTreeNode.MAX_LEAF_SIZE)
98 {
99 // No leaf node.
100 bool isVertical = treeNode.maxX - treeNode.minX > treeNode.maxY - treeNode.minY;
101 float splitValue = 0.5f * (isVertical ? treeNode.maxX + treeNode.minX : treeNode.maxY + treeNode.minY);
102
103 int left = begin;
104 int right = end;
105
106 while (left < right)
107 {
108 while (left < right && (isVertical ? m_inputObstacles[left].pos.x : m_inputObstacles[left].pos.y) < splitValue)
109 {
110 ++left;
111 }
112
113 while (right > left && (isVertical ? m_inputObstacles[right - 1].pos.x : m_inputObstacles[right - 1].pos.y) >= splitValue)
114 {
115 --right;
116 }
117
118 if (left < right)
119 {
120 ObstacleVertexData tempAgent = m_inputObstacles[left];
121 m_inputObstacles[left] = m_inputObstacles[right - 1];
122 m_inputObstacles[right - 1] = tempAgent;
123 ++left;
124 --right;
125 }
126 }
127
128 int leftSize = left - begin;
129
130 if (leftSize == 0)
131 {
132 ++leftSize;
133 ++left;
134 ++right;
135 }
136
137 treeNode.left = node + 1;
138 treeNode.right = node + 2 * leftSize;
139 m_outputTree[node] = treeNode;
140
141 BuildAgentTreeRecursive(begin, left, treeNode.left);
142 BuildAgentTreeRecursive(left, end, treeNode.right);
143 }
144 }

146 private float LeftOf(float2 a, float2 b, float2 c)
147 {
148 float x1 = a.x = c.x, y1 = a.y - c.y, x2 = b.x - a.x, y2 = b.y - a.y;
149 return x1 * y2 - y1 * x2;
150 }
