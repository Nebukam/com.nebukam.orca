71 if (!TryGetFirstInGroup(out m_agentProvider, true)

77 {
78 string msg = string.Format("Missing provider : Agents = {0}, Static obs = {1}, Agent KD = {2}, Static obs KD= {3}, " +

80 m_agentProvider,

83 m_staticObstacleKDTreeProvider,
84 m_dynObstaclesProvider,
85 m_dynObstacleKDTreeProvider, m_group);

87 throw new System.Exception(msg);
88 }
89
90 int agentCount = m_agentProvider.outputAgents.Length;
91 if (m_results.Length != agentCount)
92 {
93 m_results.Dispose();
94 m_results = new NativeArray<AgentDataResult>(agentCount, Allocator.Persistent);
95 }
96
97 //Agent data
98 job.m_inputAgents = m_agentProvider.outputAgents;
99 job.m_inputAgentTree = m_agentKDTreeProvider.outputTree;
100
101 //Static obstacles data
102 job.m_staticObstacleInfos = m_staticObstaclesProvider.outputObstacleInfos;
103 job.m_staticRefObstacles = m_staticObstaclesProvider.referenceObstacles;
104 job.m_staticObstacles = m_staticObstaclesProvider.outputObstacles;
105 job.m_staticObstacleTree = m_staticObstacleKDTreeProvider.outputTree;
106
107 //Dynamic obstacles data
108 job.m_dynObstacleInfos = m_dynObstaclesProvider.outputObstacleInfos;
109 job.m_dynRefObstacles = m_dynObstaclesProvider.referenceObstacles;
110 job.m_dynObstacles = m_dynObstaclesProvider.outputObstacles;
111 job.m_dynObstacleTree = m_dynObstacleKDTreeProvider.outputTree;
112
113 job.m_results = m_results;
114 job.m_timestep = delta / 0.25f;
115
